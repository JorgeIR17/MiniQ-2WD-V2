141static inline void GPIO_Interrupt_Init(uint8_t int_num, uint8_t mode)
142{
143 if(int_num == 2)
144 {
145 GPIO_Init(&DDRD, PORTD2, INPUT);
146 switch (mode)
147 {
148 case FALLING_EDGE: EICRA |= (1 << ISC21); break;
149 case RISING_EDGE: EICRA |= (1 << ISC20); break;
150 }
151 EIMSK |= (1 << INT2);
152 }
153
154 if(int_num == 3)
155 {
156 GPIO_Init(&DDRD, PORTD0, INPUT);
157 switch (mode)
158 {
159 case FALLING_EDGE: EICRA |= (1 << ISC31); break;
160 case RISING_EDGE: EICRA |= (1 << ISC30); break;
161 }
162 EIMSK |= (1 << INT3);
163 }
164}
