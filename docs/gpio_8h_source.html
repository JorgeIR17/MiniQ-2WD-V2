165static inline void GPIO_Interrupt_Init(uint8_t int_num, uint8_t mode)
166{
167 if(int_num == 2)
168 {
169 GPIO_Init(&DDRD, PORTD2, INPUT);
170 switch (mode)
171 {
172 case FALLING_EDGE: EICRA |= (1 << ISC21); break;
173 case RISING_EDGE: EICRA |= (1 << ISC20); break;
174 }
175 EIMSK |= (1 << INT2);
176 }
177 if(int_num == 3)
178 {
179 GPIO_Init(&DDRD, PORTD0, INPUT);
180 switch (mode)
181 {
182 case FALLING_EDGE: EICRA |= (1 << ISC31); break;
183 case RISING_EDGE: EICRA |= (1 << ISC30); break;
184 }
185 EIMSK |= (1 << INT3);
186 }
187}
